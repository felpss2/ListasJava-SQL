create database consultaSintomas;
use consultaSintomas;


CREATE TABLE DOENCAS (
    IDDOENCA INT NOT NULL auto_increment KEY,
    DOENCA_DESCRICAO VARCHAR(255) NOT NULL
);

CREATE TABLE SINTOMAS (
    IDSINTOMA INT NOT NULL auto_increment PRIMARY KEY,
    SINTOMA_DESCRICAO VARCHAR(255) NOT NULL
);

CREATE TABLE DOENCASSINTOMAS (
    IDDOENCASINTOMA INT NOT NULL auto_increment PRIMARY KEY,
    IDDOENCA INT,
    IDSINTOMA INT,
    FOREIGN KEY (IDDOENCA) REFERENCES DOENCAS(IDDOENCA),
    FOREIGN KEY (IDSINTOMA) REFERENCES SINTOMAS(IDSINTOMA)
);
