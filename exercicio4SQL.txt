CREATE DATABASE exercicio4;

CREATE TABLE Turma (
    turma_id INT PRIMARY KEY,
    disciplina_id INT,
    sigla VARCHAR(10),
    ano_semestre INT
);

CREATE TABLE Aluno (
    aluno_id INT PRIMARY KEY,
    nome VARCHAR(255)
);

CREATE TABLE Prova (
    turma_id INT,
    prova_numero INT,
    data DATE,
	FOREIGN KEY (turma_id) REFERENCES Turma(turma_id)
);

CREATE TABLE Nota (
    turma_id INT,
    aluno_id INT,
    prova_numero INT,
    nota DECIMAL(5,2),
	FOREIGN KEY (turma_id) REFERENCES Turma(turma_id),
	FOREIGN KEY (aluno_id) REFERENCES Aluno(aluno_id)
);
